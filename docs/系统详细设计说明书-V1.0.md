```python
@dataclass
class ParsedSQL:
    """SQL解析结果（完整字段）"""

    has_drop_database: bool = False

    has_where: bool = False

    has_into_outfile: bool = False
    has_index_hint: bool = False
    join_count: int = 0
    has_explicit_join: bool = False
    has_for_update: bool = False
    has_lock_tables: bool = False
    or_in_where: bool = False
    in_list_size: int = 0

    where_has_not_equal: bool = False                            # !=, <>, IS NULL, IS NOT NULL
```

### 1.2 解析流程

```python
class SQLParser:
    def parse(self, sql: str) -> ParsedSQL:
        """主解析入口"""
        sql = sql.strip().rstrip(';')
        # 1. sqlglot解析AST
        try:
            ast = sqlglot.parse_one(sql, dialect="mysql")
        except Exception:
            return self._fallback_parse(sql)  # 降级为正则匹配
        
        # 2. 判断SQL类型
        parsed = ParsedSQL(raw_sql=sql)
        parsed.sql_type = self._detect_sql_type(ast)
        
        # 3. 按类型分派解析
        if parsed.sql_type == "CREATE":
            self._parse_create(ast, parsed)
        elif parsed.sql_type == "ALTER":
            self._parse_alter(ast, parsed)
        elif parsed.sql_type == "SELECT":
            self._parse_select(ast, parsed)
        elif parsed.sql_type in ("INSERT", "REPLACE"):
            self._parse_insert(ast, parsed)
        elif parsed.sql_type in ("UPDATE", "DELETE"):
            self._parse_dml(ast, parsed)
        elif parsed.sql_type == "DROP":
            self._parse_drop(ast, parsed)
        
        # 4. 通用解析（WHERE、JOIN等）
        self._parse_common(ast, parsed)
        
        return parsed
```

### 1.3 关键解析方法设计

#### _parse_create — CREATE TABLE解析

```python
def _parse_create(self, ast, parsed: ParsedSQL):
    """解析CREATE TABLE语句"""
    parsed.is_create_table = True
    
    # 表名
    table_name = ast.this.this.name if hasattr(ast.this, 'this') else str(ast.this)
    parsed.tables = [table_name]
    parsed.table_name_plural = self._check_plural(table_name)
    
    # 列定义与索引定义
    for expr in ast.this.expressions if hasattr(ast.this, 'expressions') else []:
        if isinstance(expr, exp.ColumnDef):
            col_info = self._parse_column_def(expr)
            parsed.column_definitions.append(col_info)
            if col_info.get('comment'):
                parsed.column_comments[col_info['name']] = col_info['comment']
        elif isinstance(expr, exp.Index):
            idx_info = self._parse_index_def(expr)
            parsed.index_definitions.append(idx_info)
    
    # 表属性
    props = ast.args.get("properties")
    if props:
        for prop in props.expressions:
            if isinstance(prop, exp.EngineProperty):
                parsed.engine = str(prop.this)
            elif isinstance(prop, exp.CharacterSetProperty):
                parsed.charset = str(prop.this)
            elif "PRIMARY KEY" in str(prop):
                parsed.has_primary_key = True
            elif isinstance(prop, exp.Comment):
                parsed.has_table_comment = True
    
    # CREATE TABLE ... SELECT
    if ast.args.get("expression"):
        parsed.is_create_table_select = True
    
    # TEMPORARY
    if ast.args.get("temporary"):
        parsed.is_temporary_table = True
```

#### _parse_alter — ALTER TABLE解析

```python
def _parse_alter(self, ast, parsed: ParsedSQL):
    """解析ALTER TABLE语句，检测列长度变化和分片键修改"""
    parsed.tables = [str(ast.this)]
    
    for action in ast.expressions:
        action_info = {"action": "", "column": "", "old_type": "", "new_type": ""}
        
        if isinstance(action, exp.AlterColumn):
            action_info["action"] = "modify"
            action_info["column"] = str(action.this.this)
            action_info["new_type"] = str(action.args.get("dtype", ""))
        elif isinstance(action, exp.RenameColumn):
            action_info["action"] = "rename"
            action_info["column"] = str(action.this.this)
        
        parsed.alter_actions.append(action_info)
```

```python
class RuleChecker:
    """规则检查器"""
    
    def __init__(self, rules_config_path: str = None):
        self._rules: list[BaseRule] = []
        self._rules_by_category: dict[RuleCategory, list[BaseRule]] = {}
        self._rules_config = self._load_rules_config(rules_config_path)
        self._load_default_rules()
    
    def _load_rules_config(self, path: str) -> dict:
        """加载规则配置文件，控制每条规则的启用/禁用和白名单"""
        if path and Path(path).exists():
            with open(path) as f:
                return json.load(f)
        return {"rules": {}, "whitelist": []}
    
    def _load_default_rules(self):
        """加载所有规则模块"""
        from backend.engine.rules import ALL_RULE_CLASSES
        for rule_cls in ALL_RULE_CLASSES:
            rule = rule_cls()
            config = self._rules_config.get("rules", {}).get(rule.rule_id, {})
            rule.enabled = config.get("enabled", True)
            self._rules.append(rule)
            self._rules_by_category.setdefault(rule.category, []).append(rule)
    
    def audit_sql(self, sql: str, table_metadata: dict = None) -> AuditResult:
        """审核单条SQL"""
        parsed = self._parser.parse(sql)
        violations = []
        
        for rule in self._rules:
            if not rule.enabled:
                continue
            if self._is_whitelisted(rule, parsed):
                continue
            try:
                v = rule.check(parsed, table_metadata)
                if v:
                    violations.append(v)
            except Exception as e:
                logger.warning(f"规则{rule.rule_id}执行异常: {e}")
                continue
        
        violations = self._deduplicate_violations(violations)
        return AuditResult(
            sql=sql, sql_type=parsed.sql_type,
            passed=len([v for v in violations if v.severity == Severity.ERROR]) == 0,
            violations=violations
        )
    
    def _is_whitelisted(self, rule: BaseRule, parsed: ParsedSQL) -> bool:
        """检查规则是否被白名单豁免"""
        for entry in self._rules_config.get("whitelist", []):
            if entry.get("rule_id") == rule.rule_id:
                tables = entry.get("tables", [])
                if not tables or any(t in parsed.tables for t in tables):
                    return True
        return False
    
    def audit_file(self, content: str, file_path: str = "", table_metadata: dict = None) -> tuple[list[AuditResult], AuditSummary]:
        """审核文件（SQL文件或MyBatis XML）"""
        if self._is_mybatis_xml(content):
            sql_list = self._extract_sql_from_mybatis(content)
        else:
            sql_list = self._split_sql_file(content)
        
        results = [self.audit_sql(sql, table_metadata) for sql in sql_list]
        summary = self._build_summary(results)
        return results, summary
```

```python
class R023NoCreateTableSelect(BaseRule):
    """禁CREATE TABLE...SELECT"""
    rule_id = "R023"
    category = RuleCategory.DDL
    severity = Severity.ERROR
    description = "禁止使用CREATE TABLE ... SELECT语句，TDSQL分布式不支持，影响主从复制"
    spec_source = "TDSQL数据库开发规范 - DDL规范"
    
    def check(self, parsed, table_metadata=None):
        if parsed.is_create_table_select:
            return self._make_violation(
                "检测到CREATE TABLE ... SELECT语句，TDSQL分布式实例不支持此语法",
                "请拆分为CREATE TABLE + INSERT INTO ... SELECT两步执行"
            )
        return None

class R028TableMustHaveComment(BaseRule):
    """表必须有COMMENT"""
    rule_id = "R028"
    category = RuleCategory.DDL
    severity = Severity.ERROR
    description = "建表必须显式指定表级别COMMENT"
    spec_source = "TDSQL数据库开发规范 - 表设计规范"
    
    def check(self, parsed, table_metadata=None):
        if parsed.is_create_table and not parsed.has_table_comment:
            table_name = parsed.tables[0] if parsed.tables else "未知表"
            return self._make_violation(
                f"表 {table_name} 缺少表级别COMMENT",
                "请添加 COMMENT '表用途说明' 到CREATE TABLE语句中"
            )
        return None

class R035CrossTableFieldTypeConsistency(BaseRule):
    """多表同含义字段类型必须一致"""
    rule_id = "R035"
    category = RuleCategory.DDL
    severity = Severity.ERROR
    description = "多个数据表中相同业务含义字段的名称、类型、长度必须保持一致"
    spec_source = "TDSQL数据库开发规范 - 列设计规范"
    
    def check(self, parsed, table_metadata=None):
        # 需要元数据辅助：检查当前建表的列定义与已有表的同名列类型是否一致
        if not parsed.is_create_table or not table_metadata:
            return None
        # table_metadata格式: {"existing_columns": {"col_name": "varchar(32)", ...}}
        existing = table_metadata.get("existing_columns", {})
        for col in parsed.column_definitions:
            col_name = col.get("name", "")
            new_type = col.get("type", "")
            if col_name in existing and existing[col_name] != new_type:
                return self._make_violation(
                    f"字段 {col_name} 类型({new_type})与已有表中的同名字段类型({existing[col_name]})不一致",
                    f"请统一 {col_name} 的类型为 {existing[col_name]}"
                )
        return None
```

```python
class R053IndexNamingConvention(BaseRule):
    """索引命名规范"""
    rule_id = "R053"
    category = RuleCategory.INDEX
    severity = Severity.ERROR
    description = "索引名称必须遵循命名规范：主键pk_、唯一键uniq_、普通索引idx_前缀"
    spec_source = "TDSQL数据库开发规范 - 索引设计规范"
    
    def check(self, parsed, table_metadata=None):
        if not parsed.is_create_table:
            return None
        violations = []
        for idx in parsed.index_definitions:
            name = idx.get("name", "")
            idx_type = idx.get("type", "")
            if idx_type == "PRIMARY":
                expected_prefix = "pk_"
            elif idx_type == "UNIQUE":
                expected_prefix = "uniq_"
            else:
                expected_prefix = "idx_"
            if name and not name.startswith(expected_prefix):
                violations.append(self._make_violation(
                    f"索引 '{name}' 命名不符合规范，应以 '{expected_prefix}' 开头",
                    f"建议重命名为 {expected_prefix}{'_'.join(idx.get('columns', []))}"
                ))
        return violations[0] if violations else None
```

```python
class R066TransactionDmlLimit(BaseRule):
    """单事务DML≤1000行"""
    rule_id = "R066"
    category = RuleCategory.TRANSACTION
    severity = Severity.ERROR
    description = "单个事务中INSERT/UPDATE/DELETE/REPLACE语句操作的行数控制在1000行以内"

            )
```

```python
class DistributedExplainAnalyzer:
    """TDSQL分布式EXPLAIN分析器"""
    
    def analyze(self, explain_rows: list[dict], sql: str, 
                table_metadata: dict = None) -> DistributedExplainReport:
        """
        分析分布式EXPLAIN结果
        
        Args:
            explain_rows: EXPLAIN返回的行列表，每行包含
                {id, select_type, table, partitions, type, possible_keys, 
                 key, key_len, ref, rows, filtered, Extra}
            sql: 原始SQL
            table_metadata: 表元数据（含分片键信息）
        
        Returns:
            DistributedExplainReport
        """
        report = DistributedExplainReport()
        
        # 1. 分片键检测
        shard_key = table_metadata.get("shard_key") if table_metadata else None
        if shard_key:
            has_shard_key_in_where = self._check_shard_key_in_sql(sql, shard_key)
            if not has_shard_key_in_where:
                report.add_warning(
                    "shard_key_missing",
                    "WARNING",
                    f"WHERE条件未包含分片键 '{shard_key}'，SQL将广播到所有SET执行，性能急剧下降",
                    "请在WHERE条件中添加分片键字段"
                )
        
        # 2. 分区裁剪检测
        for row in explain_rows:
            partitions = row.get("partitions", "")
            if partitions and partitions != "NULL":
                if "," in partitions:
                    report.add_warning(
                        "partition_not_pruned",
                        "WARNING",
                        f"表 {row['table']} 扫描了多个分区({partitions})，分区裁剪未生效",
                        "请在WHERE条件中添加分区键的等值或范围条件"
                    )
        
        # 3. 跨SET JOIN检测
        join_tables = [r["table"] for r in explain_rows if r["table"] and r["table"] != "<derived>"]
        if len(join_tables) > 1 and not has_shard_key_in_where:
            report.add_warning(
                "cross_set_join",
                "WARNING",
                f"多表JOIN({', '.join(join_tables)})未带分片键，可能导致跨SET数据传输",
                "建议关联表使用相同分片键，并在JOIN条件中包含分片键"
            )
        
        # 4. 访问类型分析（复用现有逻辑）
        for row in explain_rows:
            access_type = row.get("type", "")
            if access_type == "ALL":
                report.add_warning(
                    "full_table_scan",
                    "ERROR",
                    f"表 {row['table']} 全表扫描(type=ALL)，预估扫描{row.get('rows', '?')}行",
                    "请为WHERE条件字段添加索引"
                )
        
        return report
```

```python
class FingerprintEngine:
    """SQL指纹引擎"""
    
    def fingerprint(self, sql: str) -> str:
        """
        将SQL归一化为指纹（去除具体值，保留结构）
        
        示例:
          SELECT * FROM t_order WHERE id = 1001 AND status = 'PAID'
          → SELECT * FROM t_order WHERE id = ? AND status = ?
        """
        sql = sql.strip().rstrip(';')
        
        # 1. 使用sqlglot解析
        try:
            ast = sqlglot.parse_one(sql, dialect="mysql")
            # 2. 遍历AST，将字面量替换为占位符
            for node in ast.walk():
                if isinstance(node, exp.Literal):
                    node.replace(exp.Placeholder())
                elif isinstance(node, exp.Parameter):
                    node.replace(exp.Placeholder())
            # 3. 重新生成SQL文本
            fingerprint = ast.sql(dialect="mysql")
            # 4. 统一大小写
            fingerprint = self._normalize_keywords(fingerprint)
            return fingerprint
        except Exception:
            # 降级为正则替换
            return self._regex_fingerprint(sql)
    
    def _regex_fingerprint(self, sql: str) -> str:
        """正则降级指纹生成"""
        import re
        # 替换字符串字面量
        sql = re.sub(r"'[^']*'", "?", sql)
        # 替换数字字面量
        sql = re.sub(r"\b\d+\b", "?", sql)
        # 替换IN列表
        sql = re.sub(r"IN\s*\([^)]+\)", "IN (?)", sql, flags=re.IGNORECASE)
        # 统一大小写
        return sql.upper()
    
    def aggregate_by_fingerprint(self, records: list[dict]) -> dict[str, FingerprintStats]:
        """按指纹聚合统计"""
        result = {}
        for record in records:
            fp = self.fingerprint(record["sql_text"])
            if fp not in result:
                result[fp] = FingerprintStats(
                    fingerprint=fp,
                    sample_sql=record["sql_text"],
                    exec_count=0, total_time_ms=0, max_time_ms=0,
                    rows_examined=0, rows_sent=0
                )
            stats = result[fp]
            stats.exec_count += 1
            stats.total_time_ms += record.get("total_time_ms", 0)
            stats.max_time_ms = max(stats.max_time_ms, record.get("max_time_ms", 0))
            stats.rows_examined += record.get("rows_examined", 0)
            stats.rows_sent += record.get("rows_sent", 0)
        return result
```

```python
        """
        推荐覆盖索引
        
        策略：
        1. 提取WHERE条件中的等值字段（作为索引前缀）
        2. 提取SELECT中的字段（追加到索引尾部）
        3. 提取ORDER BY字段（追加到索引尾部）

            return None

        select_cols = [c for c in self._extract_select_columns(parsed) if c not in where_cols]
        order_cols = [c for c in self._extract_orderby_columns(parsed) 
                      if c not in where_cols and c not in select_cols]
        
        index_cols = where_cols + select_cols + order_cols
        if not index_cols:
            return None
        
        index_name = f"idx_{'_'.join(c.lower() for c in index_cols[:3])}"
        ddl = f"ALTER TABLE {parsed.tables[0]} ADD INDEX {index_name} ({', '.join(index_cols)})"
        
        return IndexRecommendation(
            type="covering_index",
            table=parsed.tables[0],
            index_name=index_name,
            columns=index_cols,
            ddl=ddl,
            reason="覆盖索引可避免回表查询，提升SELECT性能"
        )
    
    def recommend_composite_index(self, parsed: ParsedSQL,
                                   table_metadata: dict) -> Optional[IndexRecommendation]:
        """
        推荐复合索引字段顺序
        
        策略：等值查询字段在前 → 范围查询字段在后 → 排序字段在尾

            type="composite_index",
            table=parsed.tables[0],
            index_name=index_name,
            columns=index_cols,
            ddl=ddl,
            reason=f"等值字段({eq_cols})在前，范围字段({range_cols})在后，排序字段({sort_cols})在尾"
```

```python
               has_delete_history: bool = False,
               has_replication_lag: bool = False,
               is_distributed: bool = False) -> PartitionAdvice:
        """按6条件判定矩阵生成分区改造建议"""
        advices = []
        
        if table_info.size_gb >= 10:
            advices.append(("①数据量≥10GB", "时间维度：RANGE BY 月/天；非时间维度：HASH BY 主键"))
        
        if table_info.rows >= 30_000_000:
            advices.append(("②行数≥3000万", "RANGE BY 时间 + HASH BY 业务键（二级分区）"))
        
        if table_info.daily_increment_gb and table_info.daily_increment_gb >= 1:
            advices.append(("③日增量≥1GB", "RANGE BY 天分区"))
        
        if has_delete_history:
            advices.append(("④有DELETE历史数据需求", "RANGE BY 时间，DELETE改为DROP PARTITION"))
        
        if has_replication_lag:
            advices.append(("⑤曾因DELETE大事务引发主备延迟", "RANGE BY 时间（强制）"))
        
        if is_distributed and table_info.size_gb >= 10:
            advices.append(("⑥TDSQL分布式+单SET表≥10GB", "已分片 + 一级RANGE时间分区"))
        
        # 生成DDL示例
        ddl_example = self._generate_partition_ddl(table_info, advices)
        
        return PartitionAdvice(
            table=f"{table_info.schema}.{table_info.table}",
            conditions=advices,
            ddl_example=ddl_example
        )
```

## 九、质量门禁服务详细设计 (M10)

```python
class GateService:
    """质量门禁服务"""
    
    def __init__(self, gate_rules_path: str = None):
        self._gate_rules: dict[str, GateRule] = {}
        self._load_gate_rules(gate_rules_path)
    
    def evaluate(self, audit_results: list[AuditResult], 
                 project_id: str) -> GateResult:
        """评估审核结果是否满足门禁规则"""
        gate_rule = self._gate_rules.get(project_id)
        if not gate_rule:
            # 默认门禁规则
            gate_rule = GateRule(
                project_id=project_id,
                max_error_count=0,
                max_warning_count=-1
            )
        
        all_violations = []
        for result in audit_results:
            all_violations.extend(result.violations)
        
        error_count = sum(1 for v in all_violations if v.severity == Severity.ERROR)
        warning_count = sum(1 for v in all_violations if v.severity == Severity.WARNING)
        
        # 检查阻断规则
        blocked_by = []
        for v in all_violations:
            if v.rule_id in gate_rule.blocked_rules:
                blocked_by.append(v.rule_id)
        
        # 评估门禁
        passed = (error_count <= gate_rule.max_error_count 
                  and warning_count <= (gate_rule.max_warning_count if gate_rule.max_warning_count >= 0 else 999999)
                  and len(blocked_by) == 0
                  and all(self._check_required_rule(r, all_violations) for r in gate_rule.required_rules))
        
        return GateResult(
            passed=passed,
            gate_rule_id=project_id,
            error_count=error_count,
            warning_count=warning_count,
            blocked_by=blocked_by,
            detail=self._build_detail(error_count, warning_count, blocked_by, passed)
        )
    
    def _check_required_rule(self, rule_id: str, violations: list[Violation]) -> bool:
        """检查必须通过的规则是否通过"""
        return not any(v.rule_id == rule_id for v in violations)
```

```python
class MonitorService:
    """监控告警服务"""
    
    METRIC_QUERIES = {
        "threads_running": "SHOW GLOBAL STATUS LIKE 'Threads_running'",
        "seconds_behind_master": "SHOW SLAVE STATUS",
        "lock_wait_count": "SELECT COUNT(*) AS cnt FROM sys.innodb_lock_waits",
        "long_transaction_count": """
            SELECT COUNT(*) AS cnt FROM information_schema.innodb_trx
            WHERE TIMESTAMPDIFF(SECOND, trx_started, NOW()) > 60
        """,
    }
    
    ALERT_THRESHOLDS = {
        "threads_running": {"warning": 100, "urgent": 200},
        "seconds_behind_master": {"warning": 60, "urgent": 300},
        "lock_wait_count": {"warning": 1, "urgent": 10},
        "long_transaction_count": {"warning": 1, "urgent": 5},
    }
    
    def collect_metrics(self, connection_id: str) -> dict:
        """采集监控指标"""
        conn = self._get_connection(connection_id)
        metrics = {}
        for name, sql in self.METRIC_QUERIES.items():
            try:
                result = conn.execute(sql).fetchone()
                metrics[name] = self._extract_metric(name, result)
            except Exception as e:
                metrics[name] = None
                logger.warning(f"采集指标{name}失败: {e}")
        return metrics
    
    def check_alerts(self, metrics: dict, connection_id: str) -> list[Alert]:
        """检查阈值告警"""
        alerts = []
        for name, value in metrics.items():
            if value is None:
                continue
            thresholds = self.ALERT_THRESHOLDS.get(name, {})
            if thresholds.get("urgent") and value >= thresholds["urgent"]:
                alerts.append(Alert(
                    metric=name, value=value, level="URGENT",
                    connection_id=connection_id,
                    message=f"{name}={value}，超过紧急阈值{thresholds['urgent']}"
                ))
            elif thresholds.get("warning") and value >= thresholds["warning"]:
                alerts.append(Alert(
                    metric=name, value=value, level="WARNING",
                    connection_id=connection_id,
                    message=f"{name}={value}，超过告警阈值{thresholds['warning']}"
                ))
        return alerts
    
    def send_alert(self, alert: Alert):
        """发送告警通知"""
        # Webhook通知
        if self._webhook_url:
            self._send_webhook(alert)
        # 记录到数据库
        self._save_alert(alert)
```

```python
from cryptography.fernet import Fernet
import os, json

class SecurityService:
    """配置安全服务"""
    
    def __init__(self):
        self._key = self._get_or_create_key()
        self._fernet = Fernet(self._key)
    
    def _get_or_create_key(self) -> bytes:
        """从环境变量获取加密密钥，不存在则生成"""
        key = os.environ.get("TDSQL_ENCRYPT_KEY")
        if not key:
            key = Fernet.generate_key().decode()
            logger.warning(
                f"TDSQL_ENCRYPT_KEY环境变量未设置，已自动生成密钥。\n"
                f"请将以下密钥保存到环境变量中：\n{TDSQL_ENCRYPT_KEY}={key}"
            )
            return key.encode()
        return key.encode() if isinstance(key, str) else key
    
    def encrypt_password(self, plaintext: str) -> str:
        """加密密码"""
        return self._fernet.encrypt(plaintext.encode()).decode()
    
    def decrypt_password(self, ciphertext: str) -> str:
        """解密密码"""
        return self._fernet.decrypt(ciphertext.encode()).decode()
    
    def encrypt_connection_config(self, config: dict) -> dict:
        """加密连接配置中的密码字段"""
        encrypted = config.copy()
        if "password" in encrypted and encrypted["password"]:
            if not encrypted["password"].startswith("gAAAAA"):  # 未加密
                encrypted["password"] = self.encrypt_password(encrypted["password"])
                encrypted["_encrypted"] = True
        return encrypted
    
    def mask_password(self, config: dict) -> dict:
        """脱敏展示（API返回时使用）"""
        masked = config.copy()
        if "password" in masked:
            masked["password"] = "***"
        return masked
```
