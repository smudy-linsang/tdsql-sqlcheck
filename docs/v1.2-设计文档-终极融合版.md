## 0. 总体目标与核心指导思想

在 v1.1.0.1 版本全面实现金融级安全防御与 100% 单元测试通过率的基础上，v1.2.0.0 版本的演进聚焦于：

> **“更实用、更高效、更能够准确发现问题、提供更加舒适无缝的用户体验”**

### 核心演进方向：
1. **更实用（ToolBridge & Operations）**：整合 `tdsql-toolkit` 离线诊断工具箱，实现任务的可视化编排、后台调度与历史日志追溯。
2. **更高效（Streaming & Connector Decoupling）**：解决大文件/多 SQL 批量审核卡死问题，实现异步流式解析，并将巨型连接器拆分为轻量级的 5 大专职连接模块。
3. **更准确（TDSQL Distributed Aware AST Engine）**：深度支持 TDSQL 专属分布式语法规则，强化 Shard Key 缺失推演、全局二级索引 (GSI) 延迟警示与广播表高高危锁表防线。
4. **更舒适的用户体验（Monaco Editor & Progressive Dual-UI）**：引入 Vite + Vue 3 现代化前端与 **Monaco Editor** 智能 SQL 编辑器，提供代码错误波浪线、智能补全与格式化；同时保留单页 `index.html` 极简兜底，实现渐进式平滑无缝切换。

#### `backend/schema/migrator.py` 实现代码
```python
import hashlib
import logging
from backend.services.database import _get_connection
from backend.schema.loader import discover_schema_files

logger = logging.getLogger("tdsql.schema.migrator")

class SchemaMigrator:
    def ensure_migration_table(self, conn):
        cursor = conn.cursor()
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS schema_migrations (
                version_key VARCHAR(128) PRIMARY KEY,
                checksum VARCHAR(64) NOT NULL,
                applied_at DATETIME DEFAULT CURRENT_TIMESTAMP
            ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
        """)
        conn.commit()

    def run_migrations(self):
        conn = _get_connection()
        try:
            self.ensure_migration_table(conn)
            cursor = conn.cursor()
            cursor.execute("SELECT version_key, checksum FROM schema_migrations")
            applied = {row["version_key"]: row["checksum"] for row in cursor.fetchall()}

            for sf in discover_schema_files():
                key = f"v{sf.version}_{sf.sequence:03d}_{sf.name}"
                checksum = hashlib.sha256(sf.sql.encode("utf-8")).hexdigest()
                if key in applied:
                    if applied[key] != checksum:
                        logger.warning(f"Schema 文件 {key} 内容已被修改，Checksum 变化！")
                    continue

                logger.info(f"应用增量数据库迁移: {key}")
                for stmt in sf.sql.split(";"):
                    s = stmt.strip()
                    if s:
                        cursor.execute(s)
                cursor.execute("""
                    INSERT INTO schema_migrations (version_key, checksum)
                    VALUES (%s, %s)
                """, (key, checksum))
                conn.commit()
        finally:
            conn.close()

migrator = SchemaMigrator()
```

### 主题 2：Engine 解耦与 TDSQL 分布式检测 (C2 + E1)

#### `backend/engine/parser/tdsql_auditor.py`
```python
from dataclasses import dataclass
import sqlglot
import sqlglot.expressions as exp

@dataclass
class TDSQLFinding:
    rule_id: str
    severity: str
    message: str
    suggestion: str

class TDSQLAuditor:
    """TDSQL 特性检测：分片键缺失推演、GSI 同步延迟及广播表锁表"""

    def check_shard_key_presence(self, expression: exp.Expression, shard_keys: list[str]) -> list[TDSQLFinding]:
        findings = []
        if not shard_keys:
            return findings

        where = expression.find(exp.Where)
        if not where:
            findings.append(TDSQLFinding(
                rule_id="DIST_001",
                severity="ERROR",
                message="缺少 WHERE 条件，将触发全节点全表广播扫描！",
                suggestion=f"必须提供分片键 ({', '.join(shard_keys)}) 条件"
            ))
            return findings

        used_cols = {col.name.lower() for col in where.find_all(exp.Column)}
        if not any(sk.lower() in used_cols for sk in shard_keys):
            findings.append(TDSQLFinding(
                rule_id="DIST_002",
                severity="WARNING",
                message=f"WHERE 条件未覆盖分片键 [{', '.join(shard_keys)}]，查询将被路由广播至所有 SET 节点！",
                suggestion="建议带上分片键实现点对点精准路由。"
            ))
        return findings
```

#### `frontend_v2/src/components/MonacoEditor.vue`
```vue
<template>
  <div ref="editorContainer" class="monaco-editor-box"></div>
</template>

<script setup>
import { ref, onMounted, watch } from 'vue';
import * as monaco from 'monaco-editor';

const props = defineProps({
  modelValue: String,
  violations: Array
});
const emit = defineEmits(['update:modelValue']);
const editorContainer = ref(null);
let editorInstance = null;

onMounted(() => {
  editorInstance = monaco.editor.create(editorContainer.value, {
    value: props.modelValue || '',
    language: 'sql',
    theme: 'vs-dark',
    automaticLayout: true,
    minimap: { enabled: false }
  });

  editorInstance.onDidChangeModelContent(() => {
    emit('update:modelValue', editorInstance.getValue());
  });
});

watch(() => props.violations, (newVal) => {
  if (!editorInstance) return;
  const model = editorInstance.getModel();
  const markers = (newVal || []).map(v => ({
    startLineNumber: v.line || 1,
    startColumn: 1,
    endLineNumber: v.line || 1,
    endColumn: 100,
    message: `[${v.rule_id}] ${v.message}`,
    severity: v.severity === 'ERROR' ? monaco.MarkerSeverity.Error : monaco.MarkerSeverity.Warning
  }));
  monaco.editor.setModelMarkers(model, 'sql-check', markers);
});
</script>
```

### 主题 5：ToolBridge 工具箱与流式批处理 (C6 + E2)

#### 异步 NDJSON 流式审核 API
```python
@router.post("/api/v1/audit/batch-stream")
async def batch_audit_stream(file: UploadFile = File(...)):
    """流式异步输出批量 SQL 审核结果，防内存溢出"""
    async def stream_generator():
        content = await file.read()
        sql_text = content.decode("utf-8")
        statements = split_sql_statements(sql_text)
        for idx, stmt in enumerate(statements, 1):
            result, _ = audit_service.audit_single_sql(stmt)
            yield json.dumps({
                "index": idx,
                "passed": result.passed,
                "violations_count": len(result.violations)
            }) + "\n"
    return StreamingResponse(stream_generator(), media_type="application/x-ndjson")
```
