## 🛠️ 一键部署与升级（直接复制运行）

```bash
#!/usr/bin/env bash
# ==============================================================================
# TDSQL-SQLCheck V1.2.0.2 一键部署与升级自动化脚本
# ==============================================================================
set -e

APP_DIR="/opt/tdsql-sqlcheck"
RELEASE_TAR="/tmp/tdsql-sqlcheck-v1.2.0.2-source.tar.gz"

echo "==== [Step 1/5] 备份旧版本并停止服务 ===="
if [ -d "${APP_DIR}" ]; then
    sudo cp -a ${APP_DIR} ${APP_DIR}-backup-$(date +%Y%m%d%H%M) || true
fi
sudo systemctl stop tdsql-sqlcheck || pkill -f "uvicorn backend.main:app" || true

echo "==== [Step 2/5] 部署代码解压 ===="
sudo mkdir -p ${APP_DIR}
if [ -f "${RELEASE_TAR}" ]; then
    sudo tar -xzf ${RELEASE_TAR} -C ${APP_DIR} --strip-components=1
else
    echo "[Info] 未发现离线包，按当前代码目录发布"
fi

cd ${APP_DIR}

echo "==== [Step 3/5] 离线 Python 虚拟环境准备 ===="
if [ ! -d "venv" ]; then
    python3 -m venv venv
fi
source venv/bin/activate

if [ -d "../wheels" ]; then
    pip install --no-index --find-links=../wheels -r requirements.txt || true
else
    pip install -r requirements.txt || true
fi

echo "==== [Step 4/5] 启动 Backend 服务 ===="
nohup python -m uvicorn backend.main:app --host 0.0.0.0 --port 8000 > backend_restart.log 2>&1 &

echo "==== [Step 5/5] 健康检查验货 ===="
sleep 3
HEALTH_CHECK=$(curl -s http://127.0.0.1:8000/health || echo "FAIL")
echo "健康探针返回: ${HEALTH_CHECK}"

if [[ "${HEALTH_CHECK}" == *"1.2.0.2"* ]]; then
    echo "🎉 [Success] TDSQL SQL审核工具成功升级/部署至 V1.2.0.2！"
else
    echo "❌ [Error] 启动失败，请检查 backend_restart.log"
    cat backend_restart.log | tail -n 20
    exit 1
fi
```
